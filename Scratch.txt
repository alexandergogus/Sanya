#include <SFML/Graphics.hpp>
#include <iostream>
#include "people.h"  // Assuming your class file is named people.h

int main() {
    // Create the main window
    sf::RenderWindow window(sf::VideoMode::getDesktopMode(), "SFML Game with Character Classes");

    // Load the map texture
    sf::Texture mapTexture;
    if (!mapTexture.loadFromFile("map.png")) {
        std::cerr << "Failed to load map texture!" << std::endl;
        return -1;
    }

    // Load the object texture (main character)
    sf::Texture objectTexture;
    if (!objectTexture.loadFromFile("object.png")) {
        std::cerr << "Failed to load object texture!" << std::endl;
        return -1;
    }

    // Create the map sprite
    sf::Sprite mapSprite;
    mapSprite.setTexture(mapTexture);

    // Create the object sprite (main character)
    sf::Sprite objectSprite;
    objectSprite.setTexture(objectTexture);

    // Create main character instance
    main_character player("Hero", 100, 200);  // name, health, velocity

    // Scale the object to make it smaller
    objectSprite.setScale(0.5f, 0.5f);

    // Set the initial position of the object (center of the map)
    sf::Vector2f objectPosition(mapTexture.getSize().x / 2.0f, mapTexture.getSize().y / 2.0f);
    objectSprite.setPosition(objectPosition);

    // Set the origin of the object to its center
    objectSprite.setOrigin(objectTexture.getSize().x / 2.0f, objectTexture.getSize().y / 2.0f);

    // Define the speed of the object (based on character's velocity)
    float moveSpeed = player.velocity;  // Using character's velocity attribute

    // Define the view (camera)
    sf::View view = window.getDefaultView();

    // Track the object's current scale
    sf::Vector2f objectScale(0.5f, 0.5f);

    // Create some NPCs (for potential interactions)
    fuckers evilProf("Dr. Evil", 50, 100, 10);  // name, health, velocity, level_of_fucking
    normal_professors niceProf("Dr. Nice", 60, 80);  // name, health, velocity
    groopmates friend1("Buddy", 80, 120, 5);  // name, health, velocity, friend_points

    // Main loop
    sf::Clock clock;
    while (window.isOpen()) {
        // Handle events
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed)
                window.close();
        }

        // Calculate delta time
        float deltaTime = clock.restart().asSeconds();

        // Handle movement
        if (sf::Keyboard::isKeyPressed(sf::Keyboard::W)) {
            objectPosition.y -= moveSpeed * deltaTime;
        }
        if (sf::Keyboard::isKeyPressed(sf::Keyboard::S)) {
            objectPosition.y += moveSpeed * deltaTime;
        }
        if (sf::Keyboard::isKeyPressed(sf::Keyboard::A)) {
            objectPosition.x -= moveSpeed * deltaTime;
            objectScale.x = -0.5f;  // Flip when moving left
            objectSprite.setScale(objectScale);
        }
        if (sf::Keyboard::isKeyPressed(sf::Keyboard::D)) {
            objectPosition.x += moveSpeed * deltaTime;
            objectScale.x = 0.5f;  // Normal when moving right
            objectSprite.setScale(objectScale);
        }

        // Boundary checking
        objectPosition.x = std::max(objectPosition.x, 0.0f);
        objectPosition.x = std::min(objectPosition.x, static_cast<float>(mapTexture.getSize().x));
        objectPosition.y = std::max(objectPosition.y, 0.0f);
        objectPosition.y = std::min(objectPosition.y, static_cast<float>(mapTexture.getSize().y));

        // Update the object's position
        objectSprite.setPosition(objectPosition);

        // Update the view (camera) to center on the object
        view.setCenter(objectPosition);
        window.setView(view);

        // Clear the window
        window.clear();

        // Draw the map and the object
        window.draw(mapSprite);
        window.draw(objectSprite);

        // Display the window
        window.display();

        // Example interaction check (could be based on position)
        if (objectPosition.x < 100 && objectPosition.y < 100) {
            evilProf.interaction(&player);  // Player interacts with evil professor
        }
    }

    return 0;
}
